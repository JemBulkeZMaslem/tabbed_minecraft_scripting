target: (TARGET | STRING) | (TARGET | STRING) ATTR_BEGIN attribute_list* ATTR_END

attribute_list: attribute | attribute COMMA attribute_list

attribute: STRING ASSIGN (STRING | INTEGER | NBT)

list: ATTR_BEGIN list_items ATTR_END

list_items: generic_data | generic_data COMMA generic_data

location: number number number

rotation: number number

data_or_storage:  STRING (location | target | STRING)* | INTEGER | FLOAT | BOOLEAN | list | NBT

generic_data: (STRING | INTEGER | FLOAT | BOOLEAN | list | NBT)

data_source: STRING data_storage STRING | STRING generic_data

number: INTEGER | FLOAT

# COMMANDS

attribute: COMMAND target (STRING number* | STRING STRING (FLOAT* | STRING STRING number STRING | STRING FLOAT*))

bossbar: COMMAND STRING STRING STRING (STRING (STRING | number | BOOLEAN)*)*

clear: COMMAND target STRING INTEGER

data: COMMAND STRING data_storage ( STRING number* |
                                    NBT |
                                    STRING STRING (
                                      INTEGER STRING (data_storage STRING* | generic_data) | <-
                                      data_storage STRING |
                                      STRING (data_storage STRING* | generic_data)
                                      )
                                    )*


effect: COMMAND STRING (target (STRING (INTEGER (INTEGER BOOLEAN*)*)*)*)*

enchant: COMMAND target STRING INTEGER*

execute: COMMAND execute_command

function: COMMAND STRING

gamemode: COMMAND STRING target

give: COMMAND target STRING INTEGER*

kill: COMMAND target

list: COMAND

say: COMMAND (STRING | number | BOOLEAN| NBT | target)** # All except target will be interpreted as Strings

                      STRING STRING STRING* | #objective
                      STRING STRING (
                        STRING STRING INTEGER
                        target* |
                        target STRING (OPERATION target STRING | INTEGER) |
                        target STRING*
                      )
                    )

                     | STRING target*

                     )*

                                             )
                                    )
                         )

# EXECUTE COMMANDS

execute_command:  STRING (
                    STRING execute_command |                                    #align, anchored, as, at, in
                    (location | rotation | STRING target STRING*) execute_command | #facing, positioned, rotated
                    STRING STRING (                                                 #store
                      location STRING STRING FLOAT |       #block
                      STRING STRING |                      #bossbar
                      target STRING STRING FLOAT |         #entity, storage
                      target STRING |                  #score
                    ) execute_command |
                    STRING (                                                    #if/unless
                      location STRING |                #block
                      location location STRING STRING |    #blocks
                      STRING (                         #data
                        location STRING |  #block
                        target STRING |    #entity
                        STRING STRING          #storage
                      ) |
                      target |                     #entity
                      STRING |                         #predicate
                      target STRING (                  #score
                        COMPARISON target STRING |
                        STRING RANGE
                      )
                    ) execute_command* |
                    STRING COMMAND
                  )
